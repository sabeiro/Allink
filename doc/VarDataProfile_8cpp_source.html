00089       if(Rad2 > RapCorner){

00091    double b = Rad2*sin(theta);
00092    //double b = sqrt( SQR(Rad2) - SQR(RapCorner));
00093    Volume1 -= 4.*(Rad2*Rad2*theta - b*RapCorner);
