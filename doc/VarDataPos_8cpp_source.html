00088 int VarData::CalcnPos(double *Pos){
00089   double BlurDist[3]={Gen->Edge[0]/(double)SubDiv[0],Gen->Edge[1]/(double)SubDiv[1],Gen->Edge[2]/(double)SubDiv[2]};
00090   int v[3];
00091   for(int d=0;d<3;d++){
00092     double InvSubDiv = 1./(double)SubDiv[d];
00093     v[d] = (int)(Pos[d]/Gen->Edge[d]*SubDiv[d]);
00094     if(v[d] > SubDiv[d]){printf("Chain out of range %d>%d\n",v[d],SubDiv[d]); continue;}
00095     if(Pos[d] - v[d]*Gen->Edge[d]*InvSubDiv < .25*BlurDist[d])
00096       v[d] = 2*v[d] + 1;
00097     else if(Pos[d] - v[d]*Gen->Edge[d]*InvSubDiv > .75*BlurDist[d])
00098       v[d] = 2*v[d] + 3;
00099     else 
00100       v[d] = 2*v[d];
00101     //printf("%d\n",v[d]);
00102   }
00103   int nPos =  (v[0]*SubDiv[1] + v[1])*SubDiv[2]+v[2];
00104   return nPos;
00105 }
