499  double UnoUno=0.;double DueUno=0.;double ZeroUno=0.;double ZeroDue=0.;
500  for(int i=0;*(Px+i)<b || i<NMass;i++){
501  // printf("%.0f\t",*(Px+i));
502  if(*(Px+i)>=a){
503  Uno += *(Px+i);
