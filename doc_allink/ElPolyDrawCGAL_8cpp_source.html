122  v3.Mult(InvScaleUn);

126  //glColor4f(HueUp[p].r,HueUp[p].g,HueUp[p].b,HueUp[p].a);

128  glColor4f(0.,.0,0.,1.);
129  DrTriaContour(&v1,&v2,&v3);
130  }
131  for(int p=0;fcTrDown != dtDown.faces_end(); ++fcTrDown,p++){
